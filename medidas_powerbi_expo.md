COSTO REAL MATERIALES:
    COSTO MATERIAL TOTAL = SUM('C  MATERI'[IMPORTE $])

COSTO REAL MAQUILA:
    COSTO MAQUILA TOTAL = SUM('C  MAQUILA'[Importe $])
    KILOS UNICOS MAQUILA = SUM('C  MAQUILA'[Kilos])
    COSTO UNIT MAQUILA = DIVIDE([COSTO MAQUILA TOTAL],[KILOS UNICOS MAQUILA])

COSTO REAL OPERADOR LOGISTICO:
    COSTO OPERADOR LOGISTICO TOTAL = SUM('C  OPER LOG'[Importe $])
    Kilos NETOS Únicos OPERADOR = 
    SUMX (SUMMARIZE ( 'C  OPER LOG','C  OPER LOG'[KEY UNICO OPL],'C  OPER LOG'[Fundo],"KilosOLN", MAX('C  OPER LOG'[KILOS NETO])),[KilosOLN])
    COSTO UNIT OPERADOR LOGISTICO = DIVIDE([COSTO OPERADOR LOGISTICO TOTAL],[Kilos NETOS Únicos OPERADOR])
    ############################
    COSTO SENASA TOTAL = SUM('C  SENASA'[Total $])
    Kilos Únicos SENASA = 
    SUMX (SUMMARIZE ( 'C  SENASA','C  SENASA'[KEY UNICO SENASA],'C  SENASA'[Fundo],"KilosSENASA", MAX('C  SENASA'[Kilos])),[KilosSENASA])
    COSTO UNIT SENASA = DIVIDE([COSTO SENASA TOTAL], [Kilos Únicos SENASA])
    COSTO UNITARIO OPERADOR-SENASA = [COSTO UNIT OPERADOR LOGISTICO]+ [COSTO UNIT SENASA]

COSTO REAL TOTAL:
    COSTO UNITARIO COMEX TOTAL = [COSTO UNIT MAQUILA]+[COSTO UNIT MATERIALES]+[COSTO UNITARIO OPERADOR-SENASA]
